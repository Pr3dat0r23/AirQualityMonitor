167void MainWindow::handleNetworkReply(QNetworkReply *reply)
168{
169 if (reply->error() != QNetworkReply::NoError) {
170 showError(QString("Błąd sieciowy: %1").arg(reply->errorString()));
171 reply->deleteLater();
172 return;
173 }
174
175 QByteArray data = reply->readAll();
176 reply->deleteLater();
177
178 try {
179 json response = json::parse(data.toStdString());
180 QString url = reply->url().toString();
181
182 if (url.contains("findAll")) {
183 processStationsData(response);
184 }
185 else if (url.contains("sensors")) {
186 processSensorsData(response);
187 }
188 else if (url.contains("getData")) {
189 processMeasurementsData(response);
190 }
191 } catch (const std::exception &e) {
192 showError(QString("Błąd przetwarzania danych: %1").arg(e.what()));
193 }
194}

203void MainWindow::processStationsData(const json &data)
204{
205 ui->stationList->clear();
206 stations.clear();
207
208 QString location = ui->locationEdit->text().trimmed().toLower();
209
210 for (const auto &item : data) {
211 QString cityName = QString::fromStdString(item["city"]["name"].get<std::string>()).toLower();
212
213 if (!location.isEmpty() && !cityName.contains(location))
214 continue;
215
216 Station station;
217 station.id = item["id"].get<int>();
218 station.name = QString::fromStdString(item["stationName"].get<std::string>());
219
220 QString address = QString::fromStdString(item["city"]["name"].get<std::string>());
221 if (item["addressStreet"].is_string()) {
222 address += ", " + QString::fromStdString(item["addressStreet"].get<std::string>());
223 }
224 station.address = address;
225
226 station.lat = QString::fromStdString(item["gegrLat"].get<std::string>()).toDouble();
227 station.lon = QString::fromStdString(item["gegrLon"].get<std::string>()).toDouble();
228
229 stations.append(station);
230 ui->stationList->addItem(station.name + "\n" + station.address);
231 }
232
233 if (stations.isEmpty()) {
234 showError("Nie znaleziono stacji w podanej lokalizacji.");

267void MainWindow::processMeasurementsData(const json &data)
268{
269 measurements.clear();
270 QChart *chart = ui->chartView->chart();
271 chart->removeAllSeries();
272
273 // Usuwanie istniejących osi
274 foreach(QAbstractAxis *axis, chart->axes()) {
275 chart->removeAxis(axis);
276 delete axis;
277 }
278
279 if (!data.contains("values") || !data["values"].is_array()) {
280 showError("Otrzymano nieprawidłową strukturę danych pomiarowych");
281 return;
282 }
283
284 QLineSeries *series = new QLineSeries();
285 QVector<double> validValues;
286
287 for (const auto& item : data["values"]) {
288 try {
289 if (!item.contains("date") || !item.contains("value")) {
290 continue;
291 }
292
293 QDateTime dateTime = QDateTime::fromString(
294 QString::fromStdString(item["date"].get<std::string>()),
295 Qt::ISODate
296 );
297
298 if (!dateTime.isValid()) continue;
299
300 double value = 0;
301 bool valueValid = false;
302
303 if (item["value"].is_number()) {
304 value = item["value"].get<double>();
305 valueValid = true;
306 }
307 else if (item["value"].is_string()) {
308 QString valueStr = QString::fromStdString(item["value"].get<std::string>()).trimmed();
309 if (!valueStr.isEmpty()) {
310 bool conversionOk = false;
311 value = valueStr.toDouble(&conversionOk);
312 valueValid = conversionOk;
313 }
314 }
315
316 if (!valueValid || std::isnan(value) || std::isinf(value)) {
317 continue;
318 }
319
320 series->append(dateTime.toMSecsSinceEpoch(), value);
321 validValues.append(value);
322 measurements.append({dateTime, value});
323
324 } catch (const std::exception& e) {
325 qDebug() << "Błąd przetwarzania rekordu:" << e.what();
326 continue;
327 }
328 }
329
330 if (measurements.empty()) {
331 showError("Nie znaleziono żadnych prawidłowych danych pomiarowych");
332 return;
333 }
334
335 QDateTimeAxis *axisX = new QDateTimeAxis();
336 axisX->setFormat("dd.MM hh:mm");
337 axisX->setTitleText("Data i godzina");
338
339 QValueAxis *axisY = new QValueAxis();
340 axisY->setTitleText(currentSensor.paramCode + " [µg/m³]");
341
342 chart->addSeries(series);
343 chart->addAxis(axisX, Qt::AlignBottom);
344 chart->addAxis(axisY, Qt::AlignLeft);
345 series->attachAxis(axisX);
346 series->attachAxis(axisY);
347
348 if (!validValues.empty()) {
349 auto [minIt, maxIt] = std::minmax_element(validValues.begin(), validValues.end());
350 double avg = std::accumulate(validValues.begin(), validValues.end(), 0.0) / validValues.size();
351
352 ui->minValueLabel->setText(QString::number(*minIt, 'f', 2));
353 ui->maxValueLabel->setText(QString::number(*maxIt, 'f', 2));
354 ui->avgValueLabel->setText(QString::number(avg, 'f', 2));
355 }
356
357 chart->setTitle("Dane pomiarowe: " + currentSensor.paramName);
358 chart->legend()->setVisible(false);
359}

505void MainWindow::updateStats()
506{
507 if (measurements.isEmpty()) return;
508
509 double min = measurements.first().value;
510 double max = measurements.first().value;
511 double sum = 0;
512 QDateTime minTime, maxTime;
513
514 for (const auto &m : measurements) {
515 if (m.value < min) {
516 min = m.value;
517 minTime = m.date;
518 }
519 if (m.value > max) {
520 max = m.value;
521 maxTime = m.date;
522 }
523 sum += m.value;
524 }
525
526 double avg = sum / measurements.size();
527
528 QString stats = ui->infoText->toHtml();
529 stats += "<hr><b>Statystyki:</b><br>";
530 stats += QString("Min: %1 (%2)<br>").arg(min).arg(minTime.toString("dd.MM.yyyy hh:mm"));
531 stats += QString("Max: %1 (%2)<br>").arg(max).arg(maxTime.toString("dd.MM.yyyy hh:mm"));
532 stats += QString("Średnia: %1").arg(avg);
533
534 ui->infoText->setHtml(stats);
535}
